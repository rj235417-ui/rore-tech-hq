# Hardcoded-secrets audit
# Patterns for API keys, tokens, and credentials that should never
# be committed to a repo. Catches the failure mode where a key is
# pasted as a workaround and not rotated back to env vars.
# Run against: source code on staged diff (pre-commit) AND the full
# repo (pre-submission).
#
# Format: one regex per line. # marks a comment. Empty lines ignored.
#
# Lesson trace: xagent-LL-006 (hardcoded keys workaround never undone),
# xagent-LL-007 (real keys pasted into chat code examples).

# Anthropic API keys — distinctive prefix
sk-ant-[a-zA-Z0-9]{20,}

# OpenAI API keys
sk-[a-zA-Z0-9]{40,}
sk-proj-[a-zA-Z0-9]{40,}

# X / Twitter API keys and tokens
# Bearer tokens (long base64-like strings)
[A-Za-z0-9%]{100,}.{0,5}AAAAAAAAAAAA

# AWS access keys and secret keys
AKIA[0-9A-Z]{16}
ASIA[0-9A-Z]{16}
aws_secret_access_key

# Google API keys
AIza[0-9A-Za-z\-_]{35}

# Stripe keys
sk_live_[0-9a-zA-Z]{24,}
pk_live_[0-9a-zA-Z]{24,}
sk_test_[0-9a-zA-Z]{24,}
pk_test_[0-9a-zA-Z]{24,}

# Firebase service account JSON detection
"type":\s*"service_account"
"private_key":\s*"-----BEGIN

# Generic patterns — high signal for hardcoded credentials in JS/TS
# (Some false positives expected; review hits manually.)
api[_-]?key\s*[:=]\s*["'][a-zA-Z0-9]{20,}
secret\s*[:=]\s*["'][a-zA-Z0-9]{20,}
token\s*[:=]\s*["'][a-zA-Z0-9]{40,}
authorization\s*[:=]\s*["']Bearer\s

# Private keys (any provider)
-----BEGIN PRIVATE KEY-----
-----BEGIN RSA PRIVATE KEY-----
-----BEGIN OPENSSH PRIVATE KEY-----
-----BEGIN EC PRIVATE KEY-----

# Database connection strings
postgres://[^:]+:[^@]+@
mysql://[^:]+:[^@]+@
mongodb(\+srv)?://[^:]+:[^@]+@
